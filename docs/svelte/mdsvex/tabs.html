<script>
  const tabs = ["Tab One", "Tab Two"];
	let selectedTab = "Tab One";
</script>

<div class='tabs'>
  {#each tabs as tab}
  <div class='tab' 
    on:click={() => selectedTab = tab}
    class:selectedTab={selectedTab === tab}>{tab}</div>
  <div class='spacer'>|</div>
  {/each}
</div>


{#if selectedTab === "Tab One"}

    <!-- content here -->

{:else if selectedTab === "Tab Two"}

    <!-- content here -->

{/if }
